# 多选框 Checkbox

一组备选项中进行多选

# 示例

多选框:

多选框组:

点击查看代码 
 <template>
     <div class="checkbox">
         <div>
             <h3>多选框:</h3>
         </div>
         <pf-checkbox v-model="checked">选中</pf-checkbox>
         <div>
             <h3>多选框组:</h3>
         </div>
         <pf-checkbox-group v-model="hobby">
             <pf-checkbox label="抽烟"></pf-checkbox>
             <pf-checkbox label="喝酒"></pf-checkbox>
             <pf-checkbox label="烫头"></pf-checkbox>
         </pf-checkbox-group>
 
     </div>
 </template>
 
 <script>
     export default {
         name: "my-checkbox",
         data(){
             return{
                 hobby: ['抽烟', '喝酒', '烫头'],
             }
         }
     }
 </script>

参数名称 参数描述 参数类型 默认值
v-model 双向绑定 布尔类型 false
label 单选框和value值 string，num，Boolean ' '
name name属性 string ‘ ’
